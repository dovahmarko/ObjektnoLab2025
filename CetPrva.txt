#include <iostream>

int& at(int arr[], int index) {
return arr[index];
}

int main() {
const int size = 5;
int niz[size] = {1, 2, 3, 4, 5};

std::cout << "Originalni niz: ";
for (int i = 0; i < size; i++) {
std::cout << niz[i] << " ";
}
std::cout << std::endl;

at(niz, 2) += 1;

std::cout << "Niz nakon promjene: ";
for (int i = 0; i < size; i++) {
std::cout << niz[i] << " ";
}
std::cout << std::endl;

at(niz, 0)++;

std::cout << "Niz nakon promjene: ";
for (int i = 0; i < size; i++) {
std::cout << niz[i] << " ";
}
std::cout << std::endl;

return 0;
}
